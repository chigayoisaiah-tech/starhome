    <script>
        const pricing = {
            cleaningTypes: {
                standard: 800,
                deep: 1500,
                move: 2000,
                office: 1000,
                construction: 2500
            },
            frequency: {
                oneTime: 1,
                weekly: 0.85,
                biweekly: 0.90,
                monthly: 0.95
            },
            perBedroom: 200,
            perBathroom: 150
        };

        function calculatePrice() {
            const cleaningType = document.getElementById('cleaningType').value;
            const frequency = document.querySelector('input[name="frequency"]:checked').value;
            const bedrooms = parseInt(document.getElementById('bedrooms').value) || 0;
            const bathrooms = parseInt(document.getElementById('bathrooms').value) || 0;

            if (!cleaningType) {
                document.getElementById('priceAmount').textContent = 'R0';
                return;
            }

            let basePrice = pricing.cleaningTypes[cleaningType];
            let roomsPrice = (bedrooms * pricing.perBedroom) + (bathrooms * pricing.perBathroom);
            let subtotal = basePrice + roomsPrice;
            let frequencyMultiplier = pricing.frequency[frequency];
            let total = Math.round(subtotal * frequencyMultiplier);

            document.getElementById('priceAmount').textContent = 'R' + total;
        }

        function submitBooking() {
            const cleaningType = document.getElementById('cleaningType').value;
            const cleaningTypeName = document.getElementById('cleaningType').options[document.getElementById('cleaningType').selectedIndex].text;
            const frequency = document.querySelector('input[name="frequency"]:checked').value;
            const frequencyName = document.querySelector('input[name="frequency"]:checked').nextElementSibling.textContent;
            const bedrooms = document.getElementById('bedrooms').value;
            const bathrooms = document.getElementById('bathrooms').value;
            const name = document.getElementById('name').value;
            const email = document.getElementById('email').value;
            const phone = document.getElementById('phone').value;
            const address = document.getElementById('address').value;
            const date = document.getElementById('date').value;
            const estimatedPrice = document.getElementById('priceAmount').textContent;

            if (!cleaningType || !name || !email || !phone || !address || !date) {
                alert('Please fill in all required fields');
                return;
            }

            const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
            if (!emailRegex.test(email)) {
                alert('Please enter a valid email address');
                return;
            }

            // Create WhatsApp message
            const message = `*New Booking Request - Star Home Cleaning*%0A%0A` +
                `*Customer Details:*%0A` +
                `Name: ${encodeURIComponent(name)}%0A` +
                `Phone: ${encodeURIComponent(phone)}%0A` +
                `Email: ${encodeURIComponent(email)}%0A` +
                `Address: ${encodeURIComponent(address)}%0A%0A` +
                `*Service Details:*%0A` +
                `Cleaning Type: ${encodeURIComponent(cleaningTypeName)}%0A` +
                `Frequency: ${encodeURIComponent(frequencyName)}%0A` +
                `Bedrooms: ${bedrooms}%0A` +
                `Bathrooms: ${bathrooms}%0A` +
                `Preferred Date: ${date}%0A%0A` +
                `*Estimated Price: ${estimatedPrice}*%0A%0A` +
                `Please confirm availability for this booking.`;

            // WhatsApp number (replace with actual number)
            const whatsappNumber = '27832271952'; // Format: country code + number (no + or spaces)
            const whatsappURL = `https://wa.me/${whatsappNumber}?text=${message}`;

            // Open WhatsApp

            // Show success modal
